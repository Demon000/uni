<fsm_text> ::= <item>

<item> ::= <state> | <transition> | <state> <item> | <transition> <item>

<state> ::= "s," <state_name>
<state> ::= "s," <state_name> ",f"
<state_name> ::= <alphanum> <state_name> | <alphanum>

<transition> ::= "t," <state_name> "," <state_name> "," <transition_characters>

<transition_characters> ::= <character> <transition_characters> | <range> <transition_characters> | <character> | <range>

<transition_characters> ::= <character> "-" <character>
<character> ::= anything

<alphanum> ::= '0' | '1' | ... | '9' | 'a' | ... | 'z' | 'A' | ... | 'Z'
<number> ::= <first_digit> <following_digits> | <following_digit>
<following_digits> ::= <following_digit> <following_digits> | <following_digit>
<first_digit> ::= '1' | '2' | ... | '9'
<following_digit> ::= '0' | '1' | ... | '9'
